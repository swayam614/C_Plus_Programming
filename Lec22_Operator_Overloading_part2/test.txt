#include <iostream>
using namespace std;

class TV
{
private:
int price;

public:
void setPrice(int price)
{
this->price = price;
}
int getPrice()
{
return this->price;
}
friend class Fridge;
};

class Fridge
{
private:
int price;

public:
void setPrice(int price)
{
this->price = price;
}
int getPrice()
{
return this->price;
}

int operator<(TV &t)
{
return this->price < t.price;
}

int operator>(TV &t)
{
return this->price > t.price;
}

int operator==(TV &t)
{
return this->price == t.price;
}
};

int main()
{

Fridge f;
TV t;

f.setPrice(10000);
t.setPrice(10000);

if (f < t)
cout << "Fridge is less costly than TV";
else if (f > t)
cout << "Fridge is more costly than TV";
else if (f == t)
cout << "Fridge price and TV price are eqaul";

return 0;
}
